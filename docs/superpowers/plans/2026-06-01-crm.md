# CRM (Clientes) — Plan de implementación

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Añadir un módulo de Clientes que agrupa proyectos, con ficha de contacto, roll-up de proyectos/tareas/notas y (PR2) analítica de tiempo agregado por cliente.

**Architecture:** Entidad `Client` nueva + `projects.client_id` nullable (1 cliente → N proyectos). Tareas/notas/tiempo del cliente se obtienen a través de sus proyectos (`whereIn project_id`). La agregación de tiempo reutiliza el patrón de `ReportsService` (suma `time_blocks` + `manual_entries` con `(julianday(ends_at)-julianday(starts_at))*1440`), acotada a los proyectos del cliente, en un `ClientService` dedicado.

**Tech Stack:** Laravel 11 (Blade, Eloquent, SQLite), Pest/PHPUnit feature tests (sin factories — se usa `Model::create`), Tailwind, Vite.

**Verificación global:** `php artisan migrate` aplica; `php artisan test` en verde; `npm run build` sin errores. Todas las rutas relativas a `/var/www/html/trackActivity/`, comandos desde `dashboard/`.

**Rama:** `paulo-crm` (creada desde main). Split:
- **PR1 (Tasks 1–7):** entidad + CRUD + ficha (sin tiempo) + nav + flag + asignación.
- **PR2 (Tasks 8–10):** `ClientService` + tiempo en ficha e índice.

namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Relations\HasMany;

class Client extends Model

> Nota: `nullOnDelete` actúa sobre borrado físico de la fila padre. Con SoftDeletes la fila no se borra, así que `client_id` NO se anula automáticamente al archivar. Para cumplir el test, el `destroy` del controlador desvincula los proyectos explícitamente (ver Step 3).

- [ ] **Step 3: Crear `ClientController.php`**

```php
<?php

namespace App\Http\Controllers;

use App\Models\Client;
use Illuminate\Http\RedirectResponse;
use Illuminate\Http\Request;
use Illuminate\View\View;

class ClientController extends Controller
{
    public function index(): View
    {
        return view('clients.index', [
            'clients' => Client::withCount('projects')->orderBy('name')->get(),
        ]);
    }

    public function create(): View
    {
        return view('clients.form', ['client' => new Client(), 'isNew' => true]);
    }

    public function store(Request $request): RedirectResponse
    {
        Client::create($this->validateClient($request));

        return redirect()->route('clients.index')->with('status', 'Cliente creado.');
    }

    public function show(Client $client): View
    {
        $projectIds = $client->projectIds();

        return view('clients.show', [
            'client'   => $client->load('projects'),
            'tasks'    => \App\Models\Task::with('project')
                ->whereIn('project_id', $projectIds)->orderBy('status')->get(),
            'notes'    => \App\Models\Note::whereIn('project_id', $projectIds)
                ->orderByDesc('updated_at')->get(),
        ]);
    }

    public function edit(Client $client): View
    {
        return view('clients.form', ['client' => $client, 'isNew' => false]);
    }

    public function update(Request $request, Client $client): RedirectResponse
    {
        $client->update($this->validateClient($request));

        return redirect()->route('clients.index')->with('status', 'Cliente actualizado.');
    }

    public function destroy(Client $client): RedirectResponse
    {
        // Desvincula sus proyectos antes de archivar (SoftDeletes no dispara
        // la FK nullOnDelete, que solo actúa en borrado físico).
        $client->projects()->update(['client_id' => null]);
        $client->delete();

        return redirect()->route('clients.index')->with('status', 'Cliente archivado.');
    }

    /** @return array<string,mixed> */
    private function validateClient(Request $request): array
    {
        return $request->validate([
            'name'    => ['required', 'string', 'max:128'],
            'company' => ['nullable', 'string', 'max:128'],
            'email'   => ['nullable', 'email', 'max:190'],
            'phone'   => ['nullable', 'string', 'max:64'],
            'website' => ['nullable', 'string', 'max:190'],
            'notes'   => ['nullable', 'string', 'max:2000'],
            'color'   => ['nullable', 'regex:/^#[0-9a-fA-F]{6}$/'],
        ], [
            'color.regex' => 'El color debe ser hex tipo #RRGGBB.',
        ]);
    }
}
```

```php
<?php

namespace App\Services;

use App\Models\Client;
use App\Models\Project;
use Carbon\CarbonImmutable;
use Illuminate\Support\Facades\DB;

/**
 * Agregación de tiempo por cliente: suma time_blocks (auto) + manual_entries
 * acotada a los proyectos del cliente, en un rango. Mismo cálculo que
 * ReportsService (julianday), pero filtrado por los proyectos del cliente.
 */
class ClientService
{
    private const DURATION = '(julianday(ends_at)-julianday(starts_at))*1440';

    /**
     * @return array{total:int, by_project:list<array{project_id:int,code:?string,name:string,color:string,minutes:int}>}
     */
    public function timeBreakdown(Client $client, CarbonImmutable $start, CarbonImmutable $end): array
    {
        $ids = $client->projectIds();
        if ($ids === []) {
            return ['total' => 0, 'by_project' => []];
        }

        $startUtc = $start->setTimezone('UTC')->format('Y-m-d H:i:s');
        $endUtc   = $end->setTimezone('UTC')->format('Y-m-d H:i:s');

        $blocks = DB::table('time_blocks')
            ->selectRaw('dominant_project_id as pid, SUM(' . self::DURATION . ') as minutes')
            ->where('status', '!=', 'idle')
            ->whereIn('dominant_project_id', $ids)
            ->where('starts_at', '>=', $startUtc)->where('starts_at', '<', $endUtc)
            ->groupBy('dominant_project_id')->get();

        $entries = DB::table('manual_entries')
            ->selectRaw('project_id as pid, SUM(' . self::DURATION . ') as minutes')
            ->whereIn('project_id', $ids)
            ->where('starts_at', '>=', $startUtc)->where('starts_at', '<', $endUtc)
            ->groupBy('project_id')->get();

        $totals = [];
        foreach ([$blocks, $entries] as $set) {
            foreach ($set as $r) {
                $totals[$r->pid] = ($totals[$r->pid] ?? 0) + (int) round((float) $r->minutes);
            }
        }

        $projects = Project::whereIn('id', array_keys($totals))->get()->keyBy('id');
        $byProject = [];
        foreach ($totals as $pid => $minutes) {
            if ($minutes <= 0) continue;
            $p = $projects->get($pid);
            $byProject[] = [
                'project_id' => $pid,
                'code'       => $p?->code,
                'name'       => $p?->name ?? '—',
                'color'      => $p?->color ?? '#94a3b8',
                'minutes'    => $minutes,
            ];
        }
        usort($byProject, fn ($a, $b) => $b['minutes'] <=> $a['minutes']);

        return ['total' => array_sum(array_column($byProject, 'minutes')), 'by_project' => $byProject];
    }
}
```
